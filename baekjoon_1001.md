import java.util.Scanner;
public class baekjoon_1001 {
    public static void main(String[] argc){
        Integer a, b;
        Scanner scanner = new Scanner(System.in);
        a = scanner.nextInt();
        b = scanner.nextInt();
        System.out.println((a-b)+"");
    }
}
